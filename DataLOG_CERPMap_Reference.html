# CERP Polynomial Plane — R Analysis
# Degree-31 / CERP-91 mapping
degree <- 31
cerp_exp <- 91

cerp <- function(a, b, t) {
  ft <- (1 - cos(pi * t)) / 2
  a + (b - a) * ft
}

poly31 <- function(x, coefs) {
  sum(coefs * x^(0:degree))
}

t_seq <- seq(0, 1, length.out = 512)
curve_data <- cerp(0, 1, t_seq)^cerp_exp
plot(t_seq, curve_data, type="l",
     col="#00FFE0", main="CERP^91")
module.d 
// D-lang — CERP Polynomial Module
module cerp_poly;

import std.math : cos, PI;

double cerp(double a, double b, double t)
{
    double ft = (1.0 - cos(PI * t)) / 2.0;
    return a + (b - a) * ft;
}

double polyMap(double x,
  double[] coefs, int deg)
{
    double acc = 0;
    foreach (i; 0 .. deg+1)
        acc += coefs[i] * x^^i;
    return acc;
}

// Φ(x,t) = polyMap^31 / cerp^91
double Phi(double x, double t,
           double[] coefs)
{
    return polyMap(x, coefs, 31)
           / (cerp(0,1,t)^^91 + 1e-9);
}

▸ UNIFIED MATHEMATICAL FORMULA — DEG³¹ / CERP⁹¹ FUSED WHOLE

CERP INTERPOLATION — BASE FUNCTION CERP(a, b, t) = a + (b − a) · [1 − cos(πt)] / 2
POLYNOMIAL DEGREE-31 — MAPPING PLANE P₃₁(x) = Σi=031 aᵢ · xⁱ
BINARY ENCODING — HEX ↔ UNICODE FUSION B(n) = HEX(n) ⊕ UNI(n) · 2⁸
FUSED SINGULAR FORMULA — Φ(x,t) Φ(x, t) = P₃₁(x)deg³¹ ────────────────── [ CERP(0,1,t)cerp⁹¹ + ε ] · B(n)
EXPANDED — ALL COMPONENTS FUSED Φ(x,t) = [ Σᵢ₌₀³¹ aᵢxⁱ ] ────────────────────────────────── { [(1−cos(πt))/2]⁹¹ + ε } · [HEX(n) ⊕ UNI(n)·2⁸]
WHERE: x ∈ ℝ = sprite position along plane · t ∈ [0,1] = normalized time/distance · aᵢ = polynomial coefficients (deg 31) · ε = 1×10⁻⁹ (guard against division by zero) · B(n) = binary encoding fusion scalar · HEX(n) = hexadecimal value of data byte n · UNI(n) = Unicode code point of glyph n.
